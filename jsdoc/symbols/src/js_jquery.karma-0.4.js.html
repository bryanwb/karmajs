 34 /**
 35 Checks if the argument (arg) is seted and if its type is 'type'.<br>
 36 1. if arg is seted: it returns toReturn if specified, otherwise it returns 
 37 	'true'
 38 2. if arg is not seted: it returns 'false'
 39 @param arg The param to check
 40 @param {Object} [type] The expeted type of 'arg'
 41 @param [toReturn] object or value to return in case 1
 42 @returns true | false | toReturn
 43 @example
 44 var msg = "hi";
 45 valid(msg); //returns true
 46 valid(msg, "String" ); //returns true
 47 valid(msg, "Number"); //returns false
 48 valid(msg, "String",false ); //returns false
 49 valid(msg123); //returns false
 50 **/
 51 var valid = function ( arg, type, toReturn ) {
 52 	if ( type ) {
 53 		if ( typeof arg === type ) {
 54 			if ( toReturn )
 55 				return toReturn;
 56 			return true;
 57 		}
 58 		return false
 59 	}
 60 	if ( typeof arg !== "undefined" ) return true;
 61 	return false;
 62 }

146 	defined strings.<br>We use karma.Gettext.js it's a modification of 
147 	<a href=http://jsgettext.berlios.de/doc/html/Gettext.html> 
148 	Gettext.js</a> from beril OS.
149 	@requires karma.Gettext.js
150 	@param {Object}  options The arguments of the Gettext constructor
151 	@returns {Function} A generic function to call Gettext functions
152 	The generic function will call a Gettext function according to the number of arguments passed
153 	<table>
154 		<tr>
155 			<th>No. params</th>
156 			<th>Parameters</th>
157 			<th>Gettext function called</th>
158 		</tr>
159 		<tr>
160 			<td>1</td>
161 			<td>msgid</td>
162 			<td>gettext</td>

169 			<td>3</td>
170 			<td>number, singular, plural</td>
171 			<td>ngettext</td>
172 		</tr>
173 		<tr>
174 			<td>4</td>
175 			<td>number, context, singular, plural</td>
176 			<td>npgettext</td>
177 		</tr>
178 	</table>

180 	var i18nWrapper = function ( options ) {
181 		var gt = new Gettext( options );
182 		if ( typeof ( gt ) === 'undefined' )
183 			throw new Error("Unable to initialize Gettext object");
184 		return (function (str1, str2, str3, str4 ) {
185 			var n, context, singular, plural;
186 			if (typeof(str4) != 'undefined') {
187 				// number, context, singular, plural
188 				return gt.npgettext(str2, str3, str4, str1);
189 			} else if (typeof(str3) != 'undefined') {
190 				// number, singular, plural
191 				return gt.ngettext(str2, str3, str1);
192 			} else if (typeof(str2) != 'undefined') {
193 				// context, msgid
194 				return gt.pgettext(str1, str2);
195 			} else if (typeof(str1) != 'undefined') {
196 				// msgid
197 				return gt.gettext(str1);
198 			} else {
199 				// nothing passed in; return blank string.
200 				// XXX: we could error here, but that may cause more harm than good.
201 				return '';
202 			}
203 		});

224 	}

227 	on language.alternatives.
228 	<p>The language file type could be: .po (Pootle) or .json (JSON). The 
229 	precedence between file types is defined according to which is defined first 
230 	in supportedLangFileTypes. By default Pootle files has precedence over JSON,
231 	files.</p>
232 	@see Karma
233 	@returns {String} The name of the language file loaded. Example: en-US.po
234 	**/
235 	var loadAlternatives = function ( ) {
236 		var loaded = undefined;
237 		var tryNext = true;
238 		//try to load the po or json language file if it exists. 
239 		//the lang order is acording to options.language.alternatives
240 		//the type (po or json or ...) is defined in supportedLangFileTypes
241 		$.each( that.language.alternatives, function ( c, lang ) {
242 			for (var i=0; i < that.supportedLangFileTypes.length 
243 			&& tryNext === true; i++) {
244 				$.ajax({
245 					url: that.paths.po +  lang + "." + 
246 						 that.supportedLangFileTypes[i].ext,
247 					cache: true,
248 					dataType: "text",
249 					async: false, //important: touch it at your own risk
250 					success: function( data, textStatus ){
251 						
252 						loaded =  lang + "." + 
253 								  that.supportedLangFileTypes[i].ext;
254 						//i18n
255 						//we pass the data so we avoid re-loading the file
256 						//creates the shorcout
257 						that.i18n.root[ that.i18n.shortcut ] =  i18nWrapper(
258 							{ 
259 								domain 	: lang, 
260 								file 	: { 
261 										  type: that.supportedLangFileTypes[i].type, 
262 										  uri: this.url, data: data 
263 										} 
264 							}
265 						);
266 						localiseContent( lang );
267 						tryNext = false;
268 					},
269 					error: function ( XHR, textStatus, errorThrown ) {
270 						//the file doesn't exist or it wasn't possible to load it
271 						tryNext = true;
272 					}
273 				});
274 				return tryNext;
275 			}
276 		});
277 		return loaded;
278 	}

324 			this.language.langCode,

326 		);
327 	}

343 	}

456 /**
457 Attaches an Object of Arrays for preloading. There are 3 valid categories (arrays): 
458 'images' for images, 'sounds' for sounds and 'videos' for video files.<br>
459 Any object within any category must have an 'id' and 'file' attributes. 
460 Additional attributes will be used as arguments to the constructor of the 
461 corresponding object.<br>
462 Any media file successfully loaded will be availabe through the 'library'.
463 The init finishes when all the valid files have finished loading or have 
464 failed on the process.<br>
465 Please note that the preloadinf will NOT start unless you call 'main' function. 
466 @param {Object} [toLoad] The Object that has the arrays for preloading.
467 @param {Array} [toLoad.images] The images 
468 @param {Array} [toLoad.sounds] The sounds 
469 @param {Array} [toLoad.videos] The videos 
470 @returns {Object} this
471 @see Karma#main
472 @example
473 var k = $.karma ({container: "#karma-main"/});
474 k.size(1200, 800);
475 k.init({
476 	images: [
477 		{id: "ball", file: "ball.png", localized : false },
478 	],
479 	sounds: [
480 		{id: "correct",  file: "correct.ogg"   },
481 	]
482 });//1 image and 1 sound attached to load
483 **/
484 Karma.prototype.init = function( toLoad ) {
485 	this.pendingToLoad = toLoad;
486 	return this; //chaining :)
487 }
488 
489 /**
490 Main function. Any Karma function call should be inside the callback function.
491 The callback function will be executed when the preloading finishes.
492 @param {Function} cb The callback funtion
493 @see Karma#init
494 **/
495 Karma.prototype.main = function ( cb ) {
496 	if ( valid( this.pendingToLoad ) ) {
497 		var that = this;
498 		var categories = ["images", "sounds", "videos" ];
499 		var counters = { "loaded":0, "error": 0 };
500 		var totalItems = 0;
501 		//get the total items
502 		for ( var i=0; i < categories.length; i++ ) {
503 			if ( valid ( this.pendingToLoad[ categories[ i ] ] ) ) {
504 				totalItems += this.pendingToLoad[ categories[ i ] ].length;
505 			}
506 		}
507 		/**
508 		callback to check if all the items were loaded or got an error when 
509 		loading
510 		**/
511 		var checkAllLoaded = function ( ev ) {
512 			if ( ev.type === "load") counters.loaded += 1;
513 			else counters.error += 1;
514 			if ( counters.loaded + counters.error === totalItems ) {
515 				if ( cb ) cb();
516 			}
517 		}
518 		
519 		for ( var i=0; i < categories.length; i++ ) {
520 			var category = categories[ i ];
521 			if ( valid ( this.pendingToLoad[ category ] ) ) {
522 				//load all the category elements
523 				var type = category.substr( 0, category.length-1 )
524 				$.each (this.pendingToLoad[ category ], function( key, config ){
525 					var id = config.id;
526 					delete config.id;
527 					//register the elements into the library
528 					that.library[ category ][ id ] =  Karma.prototype[ type ]( 
529 						config
530 					);
531 					that.library[ category ][ id ].media.addEventListener(
532 						"load",checkAllLoaded,false
533 					);
534 					that.library[ category ][ id ].media.addEventListener(
535 						"error",checkAllLoaded,false
536 					);
537 				});
538 			}
539 		}
540 	}else {
541 		if ( cb ) cb();
542 	}
543 }

629 	};
