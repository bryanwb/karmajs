203  * Shuffles an array of items randomly
204  * @param {Array} oldList of choices to be shuffled
205  * @returns {Array} newlist of choices randomly reordered

207 Karma.shuffle = function (oldList) {
208     var newList = oldList.slice(0);
209     for (var i = newList.length - 1; i > 0; i -= 1) {
210         var j = Karma.rand(0, i);
211         var t = newList[i];
212         newList[i] = newList[j];
213         newList[j] = t;
214     }
215     return newList;

229 /**
230  *  Gets the square of the Euclidian (ordinary) distance between 2 points.
231  * @param {Object} Point No. 0
232  * @param {Number} Point0.x
233  * @param {Number} Point0.y
234  * @param {Object} Point No. 1
235  * @param {Number} Point1.x
236  * @param {Number} Point1.y
237  * @returns {Number} The square of the Euclidian distance 
238  * @example
239  * 
240  * p0 = {x:0, y:1};
241  * p1 = {x:50, y:70};
242  * var d = distance2(p0, p1);
243  * 
244  */
245 Karma.distance2 = function ( p0, p1 ) {
246     return   (p1.x - p0.x) * (p1.x - p0.x) + (p1.y - p1.y) * (p1.y - p1.y); 
247 };

249 /**
250  * Gets the Euclidian (ordinary) distance between 2 points.<br>
251  * <b>Warning:</b> It's slower than distance2 function
252  * @param {Object} Point No. 0
253  * @param {Number} Point0.x
254  * @param {Number} Point0.y
255  * @param {Object} Point No. 1
256  * @param {Number} Point1.x
257  * @param {Number} Point1.y
258  * @returns {Number} The Euclidian distance 
259  * @example
260  * 
261  * p0 = {x:0, y:1};
262  * p1 = {x:50, y:70};
263  * var d = distance2(p0, p1);
264  * 
265  */
266 Karma.distance = function ( p0, p1 ) {
267 	return   Math.sqrt( this.distance2( p0, p1 ) ); 
268 };

270 /** Returns a random number within the range provided
271  * @param {Number} lower limit of the range, lowest number that can be returned
272  * @param {Number} upper limit of the range, highest number that can be returned
273  * @returns {Number} number that is >= lower and <= upper
274  * @example
275  * 
276  * var num = rand(0, 10);
277  * 
278  * //num could be 0, 1, 2, 3 ... or 10
279  * 
280  */

282   return Math.floor(Math.random() * (upper - lower + 1) + lower);

567 /** Prototype object for images

573  *  @example
574  *  kImage is the prototype object for images. This 'media' asset is loaded 
575  *  in a distinctly different way from the canvas or svg assets.    
576  *
577  */

581      * @type String
582      * @default ""
583      */

586      * @type Image
587      * @default undefined

591     _path : "",
592     //if using localized version of this image
593     _localized : false,
594     _type : "image", 
595     //initializes kImage instance with values provided by user
596     _init : function (image) {
597 	image._localized = image._localized || false;
598 	Karma._counters.total++;
599 
600 	if (image.name === undefined || image.file === undefined){
601 	    throw new Error("properties name and file have to be defined");
602 	} else {
603 	    this.name = image.name;
604 	    this.file = image.file;
605 	}
606 
607 	this.media = new Image(); 
608 	
609 	if(Karma._isLocalized(image._localized)){

613 	    this._path = Karma._assetPath + "image/";

621

645 	    },

648 	    "abort",

650 		Karma._counters.total++;
651 		that.status = "aborted";
652 		var errorMsg = "ABORT: " + that._type.toUpperCase() +
