 54 /** Checks if the current document type is set to HTML 5, throws
 55  * an error if not. Otherwise, initializes the karma object and returns
 56  * a reference to that object.
 57  * @namespace Global namespace for Karma library
 58  * @param {Object} [options={}] options for intializing Karma library
 59  * @param {String} [options.locale=''] sets current locale Not Yet Implemented
 60  * @param {Array} [options.image=[]] array of images to be converted into a collection
 61  * @param {Array} [options.audio=[]] array of audio to be converted into a collection
 62  * @param {Array} [options.video=[]] NYI array of videos to be converted into a collection
 63  * @param {Array} [options.svg=[]] array of SVG elements to be 
 64  * converted into a collection. Each SVG element must already exist in the html document
 65  * @param {Array} [options.canvas=[]] array of canvas elements 
 66  * to be converted into a collection. Each canvas element must already exist in the 
 67  * html document and width and height of each element must be set as attributes
 68  * @throws {Error} if the document type declaration is not set to HTML 5, e.g. 
 69  * <!DOCTYPE html>
 70  * @throws {Error} If any of the initialization parameters are invalid values
 71  * @returns {Object} Karma.karma -- reference to the initialized Karma library
 72  * @example
 73  * 
 74  * var k = Karma({ 
 75  *                 image: [ 
 76  *                    {name: "ninja", file: "ninja.png"}, 
 77  *                    {name: "cowboy", file: "cowboy.png"}
 78  *                         ],
 79  *                 audio: [
 80  *                    {name: "woosh", file: "woosh.ogg"},
 81  *                    {name: "yeehaw", file: "yeehaw.ogg"}
 82  *                         ],
 83  *                 video: [  //Not Yet Implemented
 84  *                    {name: "attack", file: "attack.ogv"},
 85  *                    {name: "ride", file: "ride.ogv"}
 86  *                         ]
 87  *                 canvas: [
 88  *                    {name: "ninja", domId: "ninjaCanvas"},
 89  *                    {name: "cowboy", domId: "cowboyCanvas"}
 90  *                         ],
 91  *                 svg: [ 
 92  *                    {name: "ninja", domId: "ninjaSvg"},
 93  *                    {name: "cowboy", domId: "cowboySvg"}
 94  *                         ],
 95  *                 });
 96  * Next, call the ready function with a callback to your program code
 97  * 
 98  * k.ready(function () { ... your application code . . . }                       
 99  * 
100  * after that you can access each asset like so
101  * k.image.ninja;
102  * k.svg.cowboy;
103  * k.audio.yeehaw.play();
104  * k.canvas.ninja.drawImage(k.image.ninja, 0, 0);
105  * 
106  */	
107 var Karma = exports.Karma  = function (options) {
108     Karma._isHtml5(document.doctype.nodeName);
109 
110     if ( Karma.karma._initialized === true ) {
111 	return Karma.karma;
112     } else {
113 	return Karma.karma._init(options);
114     }
115 };

120 /**This emulates the Object.create method in ecmascript 5 spec
121  * This isn't a full implementation as it doesn't support an all of Object.create's features
122  * This has the same functionality as Crockford's beget method
123  * and this primary building block for prototypal inheritance in
124  * this library
125  * @param {Object} parent that the new object's prototype should point to
126  * @returns {Object} a new object whose prototype is parent
127  * @example
128  * 
129  * var ninja = { weapon : "sword" };
130  * var ninja1 = Karma.create(ninja);
131  * ninja1.weapon === "sword"
132  */

134     function F () {};
135     F.prototype = parent;
136     return new F();

139 /** Returns a shallow copy of the passed in object
140  * @param {Object} target to be copied
141  * @returns {Object} a shallow copy of target
142  */
143 Karma.clone = function (target){
144     var copy = {};
145     for ( var i in target ) {
146 	if(target.hasOwnProperty(i)){
147 	    copy[i] = target[i];
148 	}
149     }
150     return copy;
151 };

153 /** Extends properties of the target object with those of 
154  * the source object
155  * @param {Object} target object to be extended 
156  * @param {Object} source whose properties will extend target
157  * @returns {Object} target extended by source
158  */
159 Karma.objectPlus = function (target, source){
160     for ( var i in source){
161 	if (source.hasOwnProperty(i)){
162 	    target[i] = source[i];
163 	}
164     }
165     return target;

182 //Enables function chaining for a specified list of function names
183 //IMPORTANT: use of closures here with "this" and "that" is __very__
184 //complicated here
185 Karma._makeChain = function (chainingFunctions) {
186     var that = this;
187     var _chainFunction = function ( name ){
188 	that[ name ] = function ( ){
189 	    var type = typeof this.ctx[name];
190 	    if ( type === "function") {
191 		this.ctx[ name ].apply( this.ctx, arguments );
192 	    }else if ( type === "string" ){
193 		this.ctx[ name ] = arguments[0];
194 	    }else {
195 		throw ("wtf?!: impossible to chain " + name + "!");
196 	    }
197 	    return this;
198 	};
199    };
200    
201    for (var i = 0; i < chainingFunctions.length; i++){
202        _chainFunction( chainingFunctions[ i ] );
203    }
204 };

358     /** Waits until all assets loaded(ready), then calls callback cb
359      * @param {Function} [cb] callback function
360      * @returns this
361      * @throws {Error} if Karma.karma is not initialized with the 
362      * Karma({ options }) function
363      * @example
364      * 
365      * var k = Karma({ . . . your assets here . . . });
366      * k.ready(function(){ .. your code here . . .});
367      * 
368      * your code will not be called until all assets have been loaded
369      * into collections
370      * 
371      */
372     ready : function( cb ) {
373 	var that = this;
374 	if (Karma.karma._initialized !== true){
375 	    throw new Error("Karma.karma not initialized");
376 	}
377 
378 	if (this._counters.loaded !== this._counters.total){
379 	    setTimeout(function(){ that.ready(cb);}, 5);
380 	} else if (cb) {
381 	    //hide the "Karma is loading..." message
382 	    this._statusDiv.setAttribute('style', 'display:none;');
383 
384 	     cb();
385 	} else if (!cb) {
386 	    //hide the "Karma is loading..." message
387 	    this._statusDiv.setAttribute('style', 'display:none;');
388 	    
389 	    //if no options passed, show it works message
390 	    this._showStarterMessage();
391 	}

393

396 	return this;

422     //matches 2 letter country code then optionally
423     //a dash or underscore followed by a country or language identifier
424     //i currently only allow a language identifier 2-3 chars long
425     _isValidLocale : function (locale) {
426 	var localeRegex = new RegExp('^[a-zA-Z][a-zA-Z]([-_][a-zA-z]{2,3})?$');
427 	return localeRegex.test(locale);
428     },

430     _normalizeLocale : function(locale) {
431 	var lang = "";
432 	var country = "";
433 	var divider = "";
434 
435 	lang = locale.slice(0, 2).toLowerCase();
436 	divider = "_";
437 	country = locale.slice(3, 6).toUpperCase();
438 	
439 	return locale.length > 2 ? "" + lang + divider + country : lang;
440     },

444     /**
445      * Converts a value from degrees to radians.
446      * @param {Number} angle The angle in degrees
447      * @returns {Number} The angle in radians 
448      */
449     radians : function( angle ){
450 	return ( angle / 180 ) * Math.PI;

452     /**
453      *  Gets the square of the Euclidian (ordinary) distance between 2 points.
454      * @param {Object} Point No. 0
455      * @param {Number} Point0.x
456      * @param {Number} Point0.y
457      * @param {Object} Point No. 1
458      * @param {Number} Point1.x
459      * @param {Number} Point1.y
460      * @returns {Number} The square of the Euclidian distance 
461      * @example
462      * 
463      * p0 = {x:0, y:1};
464      * p1 = {x:50, y:70};
465      * var d = distance2(p0, p1);
466      * 
467      */
468     distance2 : function ( p0, p1 ) {
469 	return   (p1.x - p0.x) * (p1.x - p0.x) + (p1.y - p1.y) * (p1.y - p1.y); 
470     },
471     /**
472      * Gets the Euclidian (ordinary) distance between 2 points.<br>
473      * <b>Warning:</b> It's slower than distance2 function
474      * @param {Object} Point No. 0
475      * @param {Number} Point0.x
476      * @param {Number} Point0.y
477      * @param {Object} Point No. 1
478      * @param {Number} Point1.x
479      * @param {Number} Point1.y
480      * @returns {Number} The Euclidian distance 
481      * @example
482      * 
483      * p0 = {x:0, y:1};
484      * p1 = {x:50, y:70};
485      * var d = distance2(p0, p1);
486      * 
487      */
488     distance : function ( p0, p1 ) {
489 	return   Math.sqrt( this.distance2( p0, p1 ) ); 
490     },
491     /** Returns a random number within the range provided
492      * @param {Number} lower limit of the range, lowest number that can be returned
493      * @param {Number} upper limit of the range, highest number that can be returned
494      * @returns {Number} number that is >= lower and <= upper
495      * @example
496      * 
497      * var num = rand(0, 10);
498      * 
499      * //num could be 0, 1, 2, 3 ... or 10
500      * 
501      */
502     rand : function ( lower, upper ){
503 		return Math.round( Math.random() * (upper - lower) + lower );
504     }

510 Karma._isLocalized = function (boolLocalized) {
511     if (typeof boolLocalized === "boolean" ) {
512 	if(boolLocalized === true && 
513 	   Karma.karma.locale === undefined){
514 	    throw new Error("You cannot localize a media asset" +
515 			    " if the global locale for Karma isn't set");
516 	} else {
517 	    return boolLocalized;
518 	}
519     } else if (typeof boolLocalized === undefined){
520 	return false;
521     } else{ 
522 	throw new Error("This is not a valid value for the localized option");
523     }
524 };

569 /** Prototypal object for images
570  *  @class This object is the prototype for images submitted to Karma in the
571  *  Karma() method
572  *  @ throws {Error} if the image asset is set to be localized but 
573  *  the global locale is not set on the Karma.karma object
574  *  @ throws {Error} if the name and file properties are not supplied
575  *  @example
576  *  kImage is the prototype object for images. This 'media' asset is loaded 
577  *  in a distinctly different way from the canvas or svg assets.    
578  *
579  */
580 Karma.kImage = 
581     {
582     /** file location of image
583      * @type String
584      * @default ""
585      */
586     file : "",
587     /** media object
588      * @type Image
589      * @default undefined 
590      */	
591     media : undefined,
592     //actual path to the file
593     _path : "",
594     //if using localized version of this image
595     _localized : false,
596     _type : "image", 
597     //initializes kImage instance with values provided by user
598     _init : function (image) {
599 	image._localized = image._localized || false;
600 	Karma.karma._counters.total++;
601 
602 	if (image.name === undefined || image.file === undefined){
603 	    throw new Error("properties name and file have to be defined");
604 	} else {
605 	    this.name = image.name;
606 	    this.file = image.file;
607 	}
608 
609 	this.media = new Image(); 
610 	
611 	if(Karma._isLocalized(image._localized)){
612 	    this._localized = image._localized;
613 	    this._path = Karma.karma._localePath + "image/";
614 	} else {
615 	    this._path = Karma.karma._assetPath + "image/";
616 	}
617 
618 	//IMPORTANT: This one magic line loads the file
619 	this.media.src = this.src = this._path + this.file;
620 	
621 	//add event handlers
622 	this._addEventHandlers();
623 
624 	
625 	return this;
626     },
627     //Adds event handlers to update the counters when 
628     //the image is successfully or unsuccessfully loaded
629     _addEventHandlers : function () {
630 	var that = this;
631 
632 	that.media.addEventListener(
633 	    "load", 
634 	    function (e) { 
635 		Karma.karma._counters.loaded++;
636 		Karma.karma._updateStatus();
637 		that.status = "loaded";}, false);
638 	
639 	that.media.addEventListener(
640 	    "error", 
641 	    function (e) { 
642 		Karma.karma._counters.errors++;
643 		that.status = "error";
644 		var errorMsg = "Error: " + that._type.toUpperCase() +
645 		    " " + that.name + " cannot be loaded."; 
646 		Karma.karma._updateStatus(errorMsg);
647 	    }, 
648 	    false);
649 	that.media.addEventListener(
650 	    "abort", 
651 	    function (e) { 
652 		Karma.karma._counters.total++;
653 		that.status = "aborted";
654 		var errorMsg = "ABORT: " + that._type.toUpperCase() +
655 		    " " + that.name + " loading was aborted."; 
656 		Karma.karma._updateStatus(errorMsg);
657 
658 	    }, false);
659     }
660     
661 };

669  *  @example

672  *  from the canvas or svg assets. They also have distinctly different
673  *  helper methods

679      * @type String

693     //initializes kAudio instance with values provided by user
694     _init : function (audio) {
695 	audio._localized = audio._localized || false;
696 	Karma.karma._counters.total++;
697 
698 	if (audio.name === undefined || audio.file === undefined){
