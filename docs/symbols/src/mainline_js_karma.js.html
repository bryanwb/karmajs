 54 /** Checks if the current document type is set to HTML 5, throws
 55  * an error otherwise, then initializes the karma object and returns
 56  * a reference to that object.
 57  * @namespace Global namespace for Karma library
 58  * @param {Object} [options={}] options for intializing Karma library
 59  * @param {String} [options.locale=''] sets current locale Not Yet Implemented
 60  * @param {Array} [options.images=[]] array of images to be converted into a collection
 61  * @param {Array} [options.sounds=[]] array of sounds to be converted into a collection
 62  * @param {Array} [options.videos=[]] array of videos to be converted into a collection
 63  * @param {Array} [options.svgs=[]] array of SVG elements to be 
 64  * converted into a collection. Each SVG element must already exist in the html document
 65  * @param {Array} [options.canvases=[]] array of canvas elements 
 66  * to be converted into a collection. Each canvas element must already exist in the 
 67  * html document and width and height of each element must be set as attributes
 68  * @returns {Object} Karma.karma -- reference to the initialized Karma library
 69  * @example
 70  * 
 71  * var k = Karma({ 
 72  *                 images: [ 
 73  *                    {name: "ninja", file: "ninja.png"}, 
 74  *                    {name: "cowboy", file: "cowboy.png"}
 75  *                         ],
 76  *                 sounds: [
 77  *                    {name: "woosh", file: "woosh.ogg"},
 78  *                    {name: "yeehaw", file: "yeehaw.ogg"}
 79  *                         ],
 80  *                 videos: [
 81  *                    {name: "attack", file: "attack.ogv"},
 82  *                    {name: "ride", file: "ride.ogv"}
 83  *                         ]
 84  *                 canvases: [
 85  *                    {name: "ninja", domId: "ninjaCanvas"},
 86  *                    {name: "cowboy", domId: "cowboyCanvas"}
 87  *                         ],
 88  *                 svgs: [ 
 89  *                    {name: "ninja", domId: "ninjaSvg"},
 90  *                    {name: "cowboy", domId: "cowboySvg"}
 91  *                         ],
 92  *                 });
 93  * Next, call the ready function with a callback to your program code
 94  * 
 95  * k.ready(function () { ... your application code . . . }                       
 96  * 
 97  * after that you can access each asset like so
 98  * k.images.ninja;
 99  * k.svgs.cowboy;
100  * k.sounds.yeehaw.play();
101  * k.canvases.ninja.drawImage(k.images.ninja, 0, 0);
102  * 
103  */	
104 var Karma = exports.Karma  = function (options) {
105     //throw error if doctype is not set to html5
106     Karma._isHtml5(document.doctype.nodeName);
107 
108     if ( Karma.karma._initialized === true ) {
109 	return Karma.karma;
110     } else {
111 	return Karma.karma._init(options);
112     }
113 };

118 /**This emulates the Object.create method in ecmascript 5 spec
119  * This isn't a full implementation as it doesn't support
120  * This has the same functionality as Crockford's beget method
121  * and this primary building block for prototypal inheritance in
122  * this library
123  * @param {Object} target that the new object's prototype should point to
124  * @returns {Object} object whose prototype points to target
125  * @example
126  * 
127  * var ninja = { weapon : "sword" };
128  * var ninja1 = Karma.create(ninja);
129  * ninja1.sword === "sword"
130  */
131 Karma.create = function (target){
132     function F () {};
133     F.prototype = target;
134     return new F();
135 };

137 /** Returns a shallow copy of the passed in object
138  * @param {Object} target to be copied
139  * @returns {Object} a shallow copy of target
140  */
141 Karma.clone = function (target){
142     var copy = {};
143     for ( var i in target ) {
144 	if(target.hasOwnProperty(i)){
145 	    copy[i] = target[i];
146 	}
147     }
148     return copy;
149 };

151 /** Extends properties of the target object with those of 
152  * the source object
153  * @param {Object} target object to be extended 
154  * @param {Object} source whose properties will extend target
155  * @returns {Object} target extended by source
156  */
157 Karma.objectPlus = function (target, source){
158     for ( var i in source){
159 	if (source.hasOwnProperty(i)){
160 	    target[i] = source[i];
161 	}
162     }
163     return target;
164 };

166 /** Creates a new object that is a prototype of the first argument
167  * then extends it with the properties of the second argument
168  * @param {Object} parent1 will be prototype of returned object
169  * @param {Object} parent2 will extend properties of returned object
170  * @returns {Object} object that whose prototype is parent1 and has 
171  * been extended with properties of parent2
172  */ 
173 Karma.copyObjectPlus = function (parent1, parent2){
174     function F () {};
175     F.prototype = parent1;
176     var G = new F();
177     return Karma.objectPlus(G, parent2);

180 //Enables function chaining for a specified list of function names
181 //IMPORTANT: use of closures here with "this" and "that" is __very__
182 //complicated here
183 Karma._makeChain = function (chainingFunctions) {
184     var that = this;
185     var _chainFunction = function ( name ){
186 	that[ name ] = function ( ){
187 	    var type = typeof this.ctx[name];
188 	    if ( type === "function") {
189 		this.ctx[ name ].apply( this.ctx, arguments );
190 	    }else if ( type === "string" ){
191 		this.ctx[ name ] = arguments[0];
192 	    }else {
193 		throw ("wtf?!: impossible to chain " + name + "!");
194 	    }
195 	    return this;
196 	};
197    };
198    
199    for (var i = 0; i < chainingFunctions.length; i++){
200        _chainFunction( chainingFunctions[ i ] );
201    }
202 };

356     /** Waits until all assets loaded, i.e. ready, then calls callback
357      * @param {Function} [cb] callback function
358      * @returns this
359      * @example
360      * 
361      * var k = Karma({ . . . your assets here . . . });
362      * k.ready(function(){ .. your code here . . .});
363      * 
364      * your code will not be called until all assets have been loaded
365      * 
366      */
367     ready : function( cb ) {
368 	var that = this;
369 	if (Karma.karma._initialized !== true){
370 	    throw new Error("Karma.karma not initialized");
371 	}
372 	
373 	if (this._counters.loaded !== this._counters.total){
374 	    setTimeout(function(){ that.ready(cb);}, 5);
375 	} else if (cb) { 
376 	    //hide that loader status
377 	    this._loaderDiv.setAttribute('style', 'display:none;');
378 	    cb();
379 	} else if (!cb) {
380 	    //if no options passed, show it works message
381 	    this._showStarterMessage();
382 	}
383 	
384 	return this;

410     //matches 2 letter country code then optionally
411     //a dash or underscore followed by a country or language identifier
412     //i currently only allow a language identifier 2-3 chars long
413     _isValidLocale : function (locale) {
414 	var localeRegex = new RegExp('^[a-zA-Z][a-zA-Z]([-_][a-zA-z]{2,3})?$');

418     _normalizeLocale : function(locale) {
419 	var lang = "";
420 	var country = "";
421 	var divider = "";
422 
423 	lang = locale.slice(0, 2).toLowerCase();
424 	divider = "_";
425 	country = locale.slice(3, 6).toUpperCase();
426 	
427 	return locale.length > 2 ? "" + lang + divider + country : lang;
428     },

440     /**
441      *  Gets the square of the Euclidian (ordinary) distance between 2 points.
442      * @param {Object} Point No. 0
443      * @param {Number} Point0.x
444      * @param {Number} Point0.y
445      * @param {Object} Point No. 1
446      * @param {Number} Point1.x
447      * @param {Number} Point1.y
448      * @returns {Number} The square of the Euclidian distance 
449      * @example
450      * 
451      * p0 = {x:0, y:1};
452      * p1 = {x:50, y:70};
453      * var d = distance2(p0, p1);
454      * 
455      */
456     distance2 : function ( p0, p1 ) {
457 	return   (p1.x - p0.x) * (p1.x - p0.x) + (p1.y - p1.y) * (p1.y - p1.y); 
458     },
459     /**
460      * Gets the Euclidian (ordinary) distance between 2 points.<br>
461      * <b>Warning:</b> It's slower than distance2 function
462      * @param {Object} Point No. 0
463      * @param {Number} Point0.x
464      * @param {Number} Point0.y
465      * @param {Object} Point No. 1
466      * @param {Number} Point1.x
467      * @param {Number} Point1.y
468      * @returns {Number} The Euclidian distance 
469      * @example
470      * 
471      * p0 = {x:0, y:1};
472      * p1 = {x:50, y:70};
473      * var d = distance2(p0, p1);
474      * 
475      */
476     distance : function ( p0, p1 ) {
477 	return   Math.sqrt( this.distance2( p0, p1 ) );

479     /** Returns a random number within the range provided

481      * @param {Number} upper limit of the range, highest number that can be returned
482      * @returns {Number} number that is >= lower and <= upper
483      * @example
484      * 
485      * var num = rand(0, 10);
486      * 
487      * num could be 0, 1, 2, 3 ... or 10
488      * 
489      */

496 /** Prototypal object for images, videos, and audio files but 
497  *  does not include svg or canvas elements
498  *  @class
499  *  @example
500  *  kMedia is the prototype object for images, sounds, and videos.
501  *  These 'media' assets are loaded in a distinctly different way
502  *  from the canvas or svg assets. They also have distinctly different
503  *  helper methods 
504  *  
505  *  You initialize the kMedia assets by passing an array of objects
506  */
507 Karma.kMedia = {
508     /** file location of asset
509      * @type String
510      * @default ""
511      */
512     file : "",
513     /** media object
514      * @type Audio|Image|Video
515      * @default undefined 
516      */	
517     media : undefined,
518     //actual path to the file
519     _path : "",
520     //if using localized version of this asset
521     _localized : false,
522     //sound, image, or video
523     _type : "", 
524     //initializes kMedia instance with values provided by user
525     _init : function (asset) {
526 	asset._localized = asset._localized || false;
527 	Karma.karma._counters.total++;
528 
529 	if (asset.name === undefined || asset.file === undefined){
530 	    throw new Error("properties name and file have to be defined");
531 	} else {
532 	    this.name = asset.name;
533 	    this.file = asset.file;
534 	}
535 
536 	//_type is a private variable used internally
537 	if (asset._type === undefined){
538 	    throw new Error("the _type property must be set. " +
539 			    "Blame the karma library authors as this is an internal value");
540 	} else {
541 	    if (Karma._isValidType(asset._type)){
542 		this._type = asset._type;
543 		switch ( this._type ) {
544 		case "image": this.media = new Image(); 
545 		    break;
546 		case "sound": this.media = new Audio(); 
547 		    break;
548 		//case "video":
549 		    //NYI
550 		    //this.media = new Video(); 
551 		    break;
552 		default: throw new Error("Media type not supported"); 
553 		}
554 
555 	    } else {
556 		throw new Error("the _type property supplied is invalid. " +
557 				"Blame the karma library authors as this is an internal value");
558 	    }
559 	}
560 	
561 	if(Karma._isLocalized(asset._localized)){
562 	    this._localized = asset._localized;
563 	    this._path = Karma.karma._localePath + 
564 		this._type + "s/";
565 	} else {
566 	    this._path = Karma.karma._assetPath +
567 		this._type + "s/";
568 	}
569 
570 	if(this._type === 'sound'){
571 	    this.media.autobuffer = true;
572 	}
573 
574 	//IMPORTANT: This one magic line loads the file
575 	this.media.src = this.src = this._path + this.file;
576 	
577 	//add event handlers
578 	this._addEventHandlers();
579 
580 	if (this._type === "sound"){
581 	    this.media.load();
582 	}
583 
584 	
585 	return this;
586     },
587     //Adds event handlers to update the counters when 
588     //the asset is successfully or unsuccessfully loaded
589     _addEventHandlers : function () {
590 	var that = this;
591 	that.media.addEventListener(
592 	    "load", 
593 	    function (e) { 
594 		Karma.karma._counters.loaded++;
595 		Karma.karma._updateStatus();
596 		that.status = "loaded";}, false);
597 	that.media.addEventListener(
598 	    "error", 
599 	    function (e) { 
600 		Karma.karma._counters.errors++;
601 		that.status = "error";
602 		var errorMsg = "Error: " + that._type.toUpperCase() +
603 		    " " + that.name + " cannot be loaded."; 
604 		Karma.karma._updateStatus(errorMsg);
605 	    }, 
606 	    false);
607 	that.media.addEventListener(
608 	    "abort", 
609 	    function (e) { 
610 		Karma.karma._counters.total++;
611 		that.status = "aborted";
612 		var errorMsg = "ABORT: " + that._type.toUpperCase() +
613 		    " " + that.name + " loading was aborted."; 
614 		Karma.karma._updateStatus(errorMsg);
615 
616 	    }, false);
617 
618     }
619     
620 };

737     //initializes object with values provides by user
738     _init: function (config) {
739 	for (var option in config){
740 	    if (config.hasOwnProperty(option)){
741 		switch (option){
742 		case "name":
743 		    this.name = config[option];
744 		    break;
745 		case "domId":
746 		    this.domId = config[option];
747 		    break;
748 		case "width":
749 		    if(!this.height){
750 			throw new Error("If you specify a width you must also" +
751 					"specify a height");
752 		    }
753 		    this.width = config[option];
754 		    break;
755 		case "height":
756 		    if(!this.width){
757 			throw new Error("If you specify a height you must also" +
758 					"specify a width");
759 		    }
760 		    this.height = parseInt(config.option, 10);
761 		    break;
762 		case "fps":
763 		    this.fps = parseInt(config.option, 10);
764 		    break;
765 		}
766 	    }
767 	}
768 	
769 	if(this.domId && document.getElementById(this.domId)){
770 	       	this.node = document.getElementById(this.domId);
771 		this.ctx = this.node.getContext('2d');
772 	} else {
773 	    throw new Error('you must specify a valid domId that' +
774 			    'is in your html page');
775 	}
776 
777 	if(!config.height && !config.width){
778 	    this.width = parseInt(this.node.getAttribute('width'), 10);
779 	    this.height = parseInt(this.node.getAttribute('height'), 10);
780 	}
781 
782 	return this;

850      * @typeof string
851      * @default ""
852      */
853     name : "",

856      * @default 0

858     width: 0,

860      * @type number
861      * @default 0
862      */
863     height: 0,
864     /** Status of element, either "loaded" or "error"
865      * @type string
866      * @default ""
867      */
868     status: "",
869     /**  Whether canvas is visible
870      * @type boolean
871      * @default true
872      */
873     visible: true,
874     /** Element ID for canvas element in html document
875      * @type String
876      * @default undefined
877      */
878     domId: undefined,
879     /** Reference to the DOM element
880      * @type DOMElement
881      * @default undefined
882      */
883     node: undefined,
884     /** Reference to the SVGDocument 
885      * @type SVGDocument
886      * @default undefined
887      */
888     doc: undefined,
889     /** Reference to the root element of the SVG Document 
890      * @type DocumentElement
891      * @default undefined
892      */
893     root: undefined,
894     _localized : undefined,
895     _chainingFunctions: [],
896     _init: function (config) {
897 	Karma.karma._counters.total++;
898 
899 	for (var option in config){
900 	    if (config.hasOwnProperty(option)){
901 		switch (option){
902 		case "name":
903 		    this.name = config[option];
904 		    break;
905 		case "domId":
906 		    this.domId = config[option];
907 		    break;
908 		case "width":
909 		    if(!this.height){
910 			throw new Error("If you specify a width you must also" +
911 					"specify a height");
912 		    }
913 		    this.width = parseInt(config[option], 10);
914 		    break;
915 		case "height":
916 		    if(!this.width){
917 			throw new Error("If you specify a height you must also" +
918 					"specify a width");
919 		    }
920 		    this.height = config[option];
921 		    break;
922 		case "fps":
923 		    this.fps = config[option];
924 		    break;
925 		}
926 	    }
927 	}
928 	
929 	if(this.domId && document.getElementById(this.domId)){
930 	       	this.node = document.getElementById(this.domId);
931 	} else {
932 	    throw new Error('you must specify a valid domId that' +
933 			    'is in your html page');
934 	}
935 
936 	if(!config.height && !config.width){
937 	    this.width = parseInt(this.node.getAttribute('width'), 10);
938 	    this.height = parseInt(this.node.getAttribute('height'), 10);
939 	}
940 
941 	var that = this;
942 	that._addEventHandlers();
943 
944 	that.doc = that.node.getSVGDocument();    
945 	//The SVG has already loaded
946 	if(that.doc){
947 	    that.root = that.doc.documentElement;
948 	    Karma.karma._counters.loaded++;
949 	    Karma.karma._updateStatus();
950 	    that.status = "loaded";
951 	}
952 		
953 	return this;
954 	
955 	
956     },
957     _addEventHandlers : function () {
958 	var that = this;
959 	that.node.addEventListener(
960 	    "load", 
961 	    function (e) { 
962 		that.doc = that.node.getSVGDocument();    
963 		that.root = that.doc.documentElement;
964 		Karma.karma._counters.loaded++;
965 		Karma.karma._updateStatus();
966 		that.status = "loaded";
967 	    }, false);
968 
969 	that.node.addEventListener(
970 	    "error", 
971 	    function (e) { 
972 		Karma.karma._counters.loaded--;
973 		Karma.karma._counters.errors++;
974 		that.status = "error";
975 		var errorMsg = "Error: " + that._type.toUpperCase() +
976 		    " " + that.name + " cannot be loaded."; 
977 		Karma.karma._updateStatus(errorMsg);
978 	    }, 
979 	    false);
980 	that.node.addEventListener(
981 	    "abort", 
982 	    function (e) { 
983 		that.status = "aborted";
984 		var errorMsg = "ABORT: " + that._type.toUpperCase() +
985 		    " " + that.name + " loading was aborted."; 
986 		Karma.karma._updateStatus(errorMsg);
987 
988 	    }, false);
989 
990     }
